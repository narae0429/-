#include <string>
#include <vector>
#include <queue>

using namespace std;

int solution(int bridge_length, int weight, vector<int> truck_weights) {
    int answer = 0;
    
    int sum = 0;
    int i = 0;
    queue<int> q;
    
    while(1){
        answer++;
        
        if(q.size()==bridge_length){
            sum -= q.front();
            q.pop();              
        }
        if(sum+truck_weights[i] <= weight){
            if(i==truck_weights.size()-1){
                answer += bridge_length;
                break;
            }
            q.push(truck_weights[i]);
            sum += truck_weights[i];
            i++;            
        } else{
            q.push(0);           
        }
    }
    
    return answer;
}
